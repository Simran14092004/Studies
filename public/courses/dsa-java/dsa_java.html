Important Array Questions

// Q: Find maximum subarray sum (Kadane's Algorithm)
// GATE CLASSIC — O(n) time
public static int maxSubarraySum(int[] arr) {
    int maxSoFar = arr[0], maxEndingHere = arr[0];
    for (int i = 1; i < arr.length; i++) {
        maxEndingHere = Math.max(arr[i], maxEndingHere + arr[i]);
        maxSoFar = Math.max(maxSoFar, maxEndingHere);
    }
    return maxSoFar;
}

// Q: Rotate array by k positions
// Reverse approach: O(n) time, O(1) space
public static void rotate(int[] arr, int k) {
    k = k % arr.length;
    reverse(arr, 0, arr.length - 1);
    reverse(arr, 0, k - 1);
    reverse(arr, k, arr.length - 1);
}

Linked Lists

// Node definition
class Node {
    int data;
    Node next;
    Node(int data) { this.data = data; }
}

// LinkedList operations
class LinkedList {
    Node head;
    
    // Insert at beginning: O(1)
    void insertFront(int data) {
        Node newNode = new Node(data);
        newNode.next = head;
        head = newNode;
    }
    
    // Detect cycle (Floyd's Algorithm) — GATE CLASSIC
    boolean hasCycle() {
        Node slow = head, fast = head;
        while (fast != null && fast.next != null) {
            slow = slow.next;
            fast = fast.next.next;
            if (slow == fast) return true;
        }
        return false;
    }
    
    // Reverse linked list: O(n) time, O(1) space
    Node reverse(Node head) {
        Node prev = null, curr = head;
        while (curr != null) {
            Node next = curr.next;
            curr.next = prev;
            prev = curr;
            curr = next;
        }
        return prev;
    }
    
    // Find middle node (fast-slow pointer)
    Node findMiddle() {
        Node slow = head, fast = head;
        while (fast != null && fast.next != null) {
            slow = slow.next;
            fast = fast.next.next;
        }
        return slow;
    }
}
GATE Question: In a singly linked list, if you delete a node (not the last), given only pointer to that node (not head), how? → Copy next node's data to current, delete next node.

// CLASSIC: Balanced parentheses
boolean isBalanced(String s) {
    Deque stack = new ArrayDeque<>();
    for (char c : s.toCharArray()) {
        if (c == '(' || c == '[' || c == '{') stack.push(c);
        else {
            if (stack.isEmpty()) return false;
            char top = stack.pop();
            if (c == ')' && top != '(') return false;
            if (c == ']' && top != '[') return false;
            if (c == '}' && top != '{') return false;
        }
    }
    return stack.isEmpty();
}

// GATE CLASSIC: Next Greater Element — O(n) with stack
int[] nextGreater(int[] arr) {
    int n = arr.length;
    int[] result = new int[n];
    Arrays.fill(result, -1);
    Deque stack = new ArrayDeque<>(); // stores indices
    for (int i = 0; i < n; i++) {
        while (!stack.isEmpty() && arr[i] > arr[stack.peek()])
            result[stack.pop()] = arr[i];
        stack.push(i);
    }
    return result;
}

Trees & Binary Search Trees

class TreeNode {
    int val;
    TreeNode left, right;
    TreeNode(int val) { this.val = val; }
}

// Tree Traversals (MUST KNOW FOR GATE)
// Inorder: Left → Root → Right (gives sorted order for BST)
void inorder(TreeNode root) {
    if (root == null) return;
    inorder(root.left);
    System.out.print(root.val + " ");
    inorder(root.right);
}

// BFS / Level Order Traversal
void levelOrder(TreeNode root) {
    if (root == null) return;
    Queue q = new LinkedList<>();
    q.offer(root);
    while (!q.isEmpty()) {
        TreeNode node = q.poll();
        System.out.print(node.val + " ");
        if (node.left != null) q.offer(node.left);
        if (node.right != null) q.offer(node.right);
    }
}

// Height of binary tree
int height(TreeNode root) {
    if (root == null) return 0;
    return 1 + Math.max(height(root.left), height(root.right));
}

// Check if BST is valid
boolean isValidBST(TreeNode root, long min, long max) {
    if (root == null) return true;
    if (root.val <= min || root.val >= max) return false;
    return isValidBST(root.left, min, root.val) 
        && isValidBST(root.right, root.val, max);
}
// Call: isValidBST(root, Long.MIN_VALUE, Long.MAX_VALUE)
GATE Fact: For a BST with n nodes: average case operations O(log n), worst case O(n) (skewed tree). AVL trees maintain O(log n) guaranteed.

// Merge Sort — O(n log n) always
void mergeSort(int[] arr, int left, int right) {
    if (left < right) {
        int mid = (left + right) / 2;
        mergeSort(arr, left, mid);
        mergeSort(arr, mid + 1, right);
        merge(arr, left, mid, right);
    }
}

// Quick Sort — Partition step
int partition(int[] arr, int low, int high) {
    int pivot = arr[high];
    int i = low - 1;
    for (int j = low; j < high; j++) {
        if (arr[j] <= pivot) {
            i++;
            int temp = arr[i]; arr[i] = arr[j]; arr[j] = temp;
        }
    }
    int temp = arr[i+1]; arr[i+1] = arr[high]; arr[high] = temp;
    return i + 1;
}

Dynamic Programming

DP solves problems by breaking them into overlapping subproblems and storing results (memoization/tabulation).

// 1. Fibonacci — Classic DP
// Naive recursive: O(2^n) — DON'T USE
// DP tabulation: O(n)
int fib(int n) {
    if (n <= 1) return n;
    int[] dp = new int[n + 1];
    dp[0] = 0; dp[1] = 1;
    for (int i = 2; i <= n; i++)
        dp[i] = dp[i-1] + dp[i-2];
    return dp[n];
}

// 2. Longest Common Subsequence (LCS) — GATE CLASSIC
int lcs(String s1, String s2) {
    int m = s1.length(), n = s2.length();
    int[][] dp = new int[m+1][n+1];
    for (int i = 1; i <= m; i++)
        for (int j = 1; j <= n; j++)
            if (s1.charAt(i-1) == s2.charAt(j-1))
                dp[i][j] = 1 + dp[i-1][j-1];
            else
                dp[i][j] = Math.max(dp[i-1][j], dp[i][j-1]);
    return dp[m][n];
}

// 3. 0/1 Knapsack — GATE CLASSIC
int knapsack(int[] weights, int[] values, int capacity) {
    int n = weights.length;
    int[][] dp = new int[n+1][capacity+1];
    for (int i = 1; i <= n; i++)
        for (int w = 0; w <= capacity; w++) {
            dp[i][w] = dp[i-1][w]; // don't take item i
            if (weights[i-1] <= w)
                dp[i][w] = Math.max(dp[i][w], 
                    values[i-1] + dp[i-1][w - weights[i-1]]);
        }
    return dp[n][capacity];
}

// 4. Longest Increasing Subsequence (LIS) — O(n²)
int lis(int[] arr) {
    int n = arr.length;
    int[] dp = new int[n];
    Arrays.fill(dp, 1);
    int maxLen = 1;
    for (int i = 1; i < n; i++) {
        for (int j = 0; j < i; j++)
            if (arr[j] < arr[i])
                dp[i] = Math.max(dp[i], dp[j] + 1);
        maxLen = Math.max(maxLen, dp[i]);
    }
    return maxLen;
}

Important GATE Questions & Solutions

Q1 [GATE 2020]: What is the number of binary trees with 3 nodes?

A: Catalan number C(3) = C(2n,n)/(n+1) = C(6,3)/4 = 20/4 = 5. Answer: 5
Q2 [GATE Classic]: Time complexity of building a heap from n elements?

A: O(n) — not O(n log n). Using the heapify-down approach on all non-leaf nodes gives O(n) total work due to the geometric series sum.
Q3 [GATE 2022]: Inorder traversal of a BST gives? Which traversal gives sorted output?

A: Inorder (Left → Root → Right) of a BST always gives elements in sorted ascending order.
Q4 [GATE Classic]: Worst case of Quick Sort and when does it occur?

A: O(n²) when the pivot is always the smallest or largest element (already sorted array with first/last pivot). Can be avoided with random pivot or median-of-three.
Q5 [GATE 2019]: What is the output when postorder traversal is applied to: root=1, left=2, right=3, 2's left=4, 2's right=5?

A: Postorder: Left → Right → Root → Output: 4, 5, 2, 3, 1
Q6 [GATE Classic]: Minimum number of comparisons to find max AND min of n elements simultaneously?

A: ⌊3n/2⌋ - 2 comparisons. Process elements in pairs: compare pair first, then compare larger with current max and smaller with current min.
Q7 [GATE 2021]: Time complexity of Dijkstra's algorithm using binary heap?

A: O((V + E) log V) where V = vertices, E = edges. With adjacency matrix: O(V²).
Q8 [GATE Classic — Hashing]: If a hash table has 7 slots and uses chaining, inserting keys 50, 700, 76, 85, 92, 73, 101 with h(k) = k mod 7, what is the average search time?

A: Calculate each: 50%7=1, 700%7=0, 76%7=6, 85%7=1, 92%7=1, 73%7=3, 101%7=3. Chains at slot 1: [50,85,92], slot 3: [73,101]. Average = (1+1+1+2+3+1+2)/7 ≈ 1.57
