import numpy as np
import matplotlib.pyplot as plt

def make_gradcam_heatmap(img_array, model, last_conv_layer_name='conv5_block3_out'):
    grad_model = tf.keras.models.Model([model.inputs], 
                                        [model.get_layer(last_conv_layer_name).output, model.output])
    with tf.GradientTape() as tape:
        conv_outputs, predictions = grad_model(img_array)
        loss = predictions[0, 0]   # pneumonia class score
    grads = tape.gradient(loss, conv_outputs)
    pooled_grads = tf.reduce_mean(grads, axis=(0,1,2))
    conv_outputs = conv_outputs[0]
    heatmap = conv_outputs @ pooled_grads[..., tf.newaxis]
    heatmap = tf.squeeze(heatmap)
    heatmap = tf.maximum(heatmap, 0) / tf.math.reduce_max(heatmap + 1e-10)
    return heatmap.numpy()
# Visualization: overlay heatmap on X-ray -> model focuses on opacities, proving clinical relevance.

🩺 P3 · CH1
U‑Net Architecture: precise lesion segmentation
Encoder (Conv+Pool) → Bottleneck → Decoder (UpConv) + Skip Connections → Pixel‑wise mask
      ══════════════════════════════════════════════════════════════════
✨ Skip connections recover spatial details lost in pooling → ideal for medical segmentation.
🏥 U‑Net implementation (key blocks)
from tensorflow.keras import layers, Model

def double_conv(x, filters):
    x = layers.Conv2D(filters, 3, padding='same', activation='relu')(x)
    x = layers.Conv2D(filters, 3, padding='same', activation='relu')(x)
    return x

def unet(input_size=(256,256,3)):
    inputs = layers.Input(input_size)
    # Encoder
    c1 = double_conv(inputs, 64); p1 = layers.MaxPool2D()(c1)
    c2 = double_conv(p1, 128); p2 = layers.MaxPool2D()(c2)
    c3 = double_conv(p2, 256); p3 = layers.MaxPool2D()(c3)
    # Bridge
    b = double_conv(p3, 512)
    # Decoder + skip
    u1 = layers.Conv2DTranspose(256, 2, strides=2, padding='same')(b)
    u1 = layers.concatenate([u1, c3])
    c4 = double_conv(u1, 256)
    u2 = layers.Conv2DTranspose(128, 2, strides=2)(c4)
    u2 = layers.concatenate([u2, c2])
    c5 = double_conv(u2, 128)
    u3 = layers.Conv2DTranspose(64, 2, strides=2)(c5)
    u3 = layers.concatenate([u3, c1])
    c6 = double_conv(u3, 64)
    outputs = layers.Conv2D(1, 1, activation='sigmoid')(c6)
    return Model(inputs, outputs)

⚙️ Custom Dice coefficient and loss
import tensorflow.keras.backend as K
def dice_coef(y_true, y_pred, smooth=1e-6):
    y_true_f = K.flatten(y_true)
    y_pred_f = K.flatten(y_pred)
    intersection = K.sum(y_true_f * y_pred_f)
    return (2. * intersection + smooth) / (K.sum(y_true_f) + K.sum(y_pred_f) + smooth)

def dice_loss(y_true, y_pred):
    return 1 - dice_coef(y_true, y_pred)

model.compile(optimizer='adam', loss=dice_loss, metrics=[dice_coef, 'accuracy'])

def gradcam_heatmap(img, model, last_conv='conv5_block3_out'):
    grad_model = tf.keras.models.Model([model.inputs], [model.get_layer(last_conv).output, model.output])
    with tf.GradientTape() as tape:
        conv_out, preds = grad_model(img[np.newaxis,...])
        loss = preds[0,0]
    grads = tape.gradient(loss, conv_out)
    pooled = tf.reduce_mean(grads, axis=(0,1,2))
    heatmap = tf.reduce_sum(tf.squeeze(conv_out) * pooled, axis=-1)
    heatmap = tf.maximum(heatmap, 0) / tf.reduce_max(heatmap + 1e-8)
    return heatmap.numpy()

    c = [landmarks[c_idx].x, landmarks[c_idx].y]
